class Solution {
  public static ArrayList<Integer> alternateSort(int[] arr) {
        ArrayList<Integer> l=new ArrayList<>();
       int fir=0;
        int lss=arr.length-1;
        Arrays.sort(arr);

        while(fir<=lss){

            if(fir==lss){


                l.add(arr[fir]);
                break;

            }
            l.add(arr[lss--]);
             l.add(arr[fir++]);
        }
        return l;
        //-+
    }
}
